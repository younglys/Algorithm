<h3>동전교환</h3>

```js
function solution(change, arr) {
  let answer = change;

  function DFS(L, sum) {
    // 기존에 출력된 최솟값 L보다 큰 값이 출력되는 불필요한 연산 제거
    if (sum > change || L > answer) return;
    if (sum === change) {
      answer = Math.min(answer, L);
    } else {
      for(let i = 0; i < arr.length; i++) {
        DFS(L+1, sum + arr[i]);
      }
    }
  }

  DFS(0, 0);
  return answer;
}

let arr = [1, 2, 5];
console.log(solution(15, arr));
```
