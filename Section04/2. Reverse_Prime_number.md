<h3>뒤집은 소수</h3>

```js
function solution(arr){
  let answer = [];
  let revArr = [];

  for(let i of arr){
    let res = Number(i.toString().split('').reverse().join(''));
    revArr.push(res);
  }
  for(let j of revArr) {
    if(j !== 1 && (j % 2 === 1 || j === 2) && (j % 3 === 2 || j % 3 === 1 || j === 3)) {
      answer.push(j);
    }
  }
  return answer;
}

let arr=[32, 55, 62, 20, 250, 370, 200, 30, 100];
console.log(solution(arr));
```
