<h3>괄호 문자 제거(스택)</h3>

```js
function solution(s) {
  let answer;
  let stack = [];
  
  for (let x of s) {
    if (x !== ')') {
      stack.push(x);
    } else {
      let index = stack.lastIndexOf('(');
      stack.splice(index);
    }
  }
  answer = stack.join('');
  return answer;
}

let str = "(A(BC)D)EF(G(H)(IJ)K)LM(N)";
console.log(solution(str));
```
