<h3>후위식 연산</h3>

```js
function calculate(a, b, oper) {
  if (oper === '+') return a + b;
  else if (oper === '-') return a - b;
  else if (oper === '*') return a * b;
  else if (oper === '/') return a / b;
}

function solution(s) {
  let answer;
  let stack = [];

  for (let x of s) {
    if (isNaN(x)) {
      let b = Number(stack.pop());
      let a = Number(stack.pop());
      stack.push(calculate(a, b, x));
    } else stack.push(x);
  }
  return stack;
}

let postfix = "352+*9-";
console.log(solution(postfix));
```
