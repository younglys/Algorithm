<h3>회문 문자열</h3>

```js
function solution(str) {
  const lower = str.toLowerCase();

  for (let i = 0; i < Math.floor(lower.length/2); i++) {
    if (lower[i] !== lower[lower.length - i - 1]) {
      return "NO";
    }
  }
  return "YES";
}

let str = "Reviver";
let str2 = "ABBA";
```
