<h3>삽입정렬</h3>

```js
function solution(arr){
  let answer = [];
  answer.push(arr[0]);

  for(let i = 1; i < arr.length; i++){
    for(let j = 0; j < answer.length; j++){
      if(arr[i] < answer[j]){
        answer.splice(j, 0, arr[i]);
        break;
      } 
    }
  }
  return answer;
}

let arr = [11, 7, 5, 6, 10, 9];
console.log(solution(arr));
```
